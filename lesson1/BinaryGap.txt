#include<vector>
#include<algorithm>
int solution(int N) {
    // write your code in C++14 (g++ 6.2.0)
    vector <int> v  ;
    while(N)
    {
        v.push_back(N%2);
        N = N>>1;
    }
    reverse(v.begin(),v.end());
    //for(auto a:v) cout<<a<<"";
    int start = 0; 
    int c=0;
    for(int i =0;i<v.size();++i)
    {
        if(v[i]==1 && start>=0)
        {
            c = max(c , i-start-1 );
            start = i;
        }
        else if (v[i]==1 && start ==0)
        {
            start = i;
        }
    }
    return c;
}

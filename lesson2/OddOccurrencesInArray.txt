#include<unordered_map>
int solution(vector<int> &A) {
    // write your code in C++14 (g++ 6.2.0)
    unordered_map <int,int> m;
    for(auto a :A) m[a] ++;
    int res ;
    for(int i = 0 ;i<A.size() ;++i)
    {
        if(m[A[i]]%2!=0) return res= A[i];
    }
    return res;
}
